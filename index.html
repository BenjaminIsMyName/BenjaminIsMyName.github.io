public static void findAllInRange(int min, int max) {
  int counter = 0;
  while (min < max) {
    if (!checkOne(min)) {
      System.out.println("Couldn't find any for "+min+"!");
      counter++;
    }
    min++;
  }
  System.out.println("\nAmount of numbers we couldn't find: "+counter);
}

public static boolean checkOne(int num) {
  boolean foundWay = false; 
  for (int i=0; i < num; i++) {
    if (putTogether(i, num))
      foundWay = true; 
  }
  return foundWay;
}

public static boolean putTogether(int from, int numToFind) { 
  int sum = from+(from+1); 
  for (int i=from+2; sum<=numToFind; sum+=i++) 
    if (sum == numToFind) {
      System.out.println("found "+numToFind+": "+from+" >>> "+i);
      return true;
    }	
  return false;
}

def check_id_number(id_number):
  if not id_number.isdigit() or len(id_number) > 9:
    return False
  sum = 0
  for index, digit in enumerate(id_number):
    if index % 2 == 0:
      sum += int(digit)
    else:
      sum += (int(digit) * 2) % 10 + int(digit) * 2 // 10
  return sum % 10 == 0
